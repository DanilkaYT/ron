// ============================================================
//  ИМПОРТЫ из CDN (исправлено)
// ============================================================
import { TelegramClient } from 'https://esm.sh/@mtcute/core@0.32.0';
import { TelegramWorker } from 'https://esm.sh/@mtcute/worker@0.32.0';

// ============================================================
//  ОСНОВНОЙ ОБРАБОТЧИК
// ============================================================
export default {
  async scheduled(event, env, ctx) {
    const cfg = CONFIG;
    const nowUTC = new Date();
    const localHours = (nowUTC.getUTCHours() + cfg.TIMEZONE_OFFSET + 24) % 24;
    const localMinutes = localHours * 60 + nowUTC.getUTCMinutes();

    // --- Сон (консолидация) в заданный час ---
    if (localHours === cfg.SLEEP_TIME && localMinutes < 5) {
      await sleepConsolidation(env, cfg);
      await env.KV_STATE.put('lastSleepTime', String(Date.now()));
    }

    // Загружаем состояния
    const lastCheckStr = await env.KV_STATE.get('lastCheckTime');
    const lastCheckTime = lastCheckStr ? parseInt(lastCheckStr, 10) : 0;

    const nextInitStr = await env.KV_STATE.get('nextInitiativeTime');
    let nextInitiativeTime = nextInitStr ? parseInt(nextInitStr, 10) : 0;

    const lastMsgIdStr = await env.KV_STATE.get('lastMsgId');
    const lastMsgId = lastMsgIdStr ? parseInt(lastMsgIdStr, 10) : 0;

    const contextStr = await env.KV_CONTEXT.get('context', 'text');
    let context = contextStr ? JSON.parse(contextStr) : [];

    const emotion = await env.KV_STATE.get('emotion') || 'neutral';
    const mood = await env.KV_STATE.get('mood') || 'спокойное';

    // --- Проверка новых сообщений (раз в час) ---
    let processedNewMessages = false;
    const now = Date.now();
    if (now - lastCheckTime >= cfg.CHECK_INTERVAL * 1000) {
      await env.KV_STATE.put('lastCheckTime', String(now));

      const client = await createTelegramClient(env, cfg);
      if (!client) return;

      try {
        const messages = await client.getMessages(cfg.ANDREY_ID, { limit: 30 });
        const newMessages = messages
          .filter(m => m.fromId && m.fromId.id === cfg.ANDREY_ID && m.id > lastMsgId)
          .sort((a, b) => a.id - b.id);

        if (newMessages.length > 0) {
          await handleNewMessages(newMessages, client, env, context, cfg, emotion, mood);
          processedNewMessages = true;
          await env.KV_STATE.put('lastSentStatus', 'replied');
          const lastNew = newMessages[newMessages.length - 1];
          await env.KV_STATE.put('lastMsgId', String(lastNew.id));
        } else {
          await checkAndReactToUnread(client, env, cfg);
        }
        await env.KV_CONTEXT.put('context', JSON.stringify(context));
      } catch (e) {
        console.error('Ошибка при проверке:', e);
      } finally {
        await client.close();
      }
    }

    // --- Инициатива ---
    if (!processedNewMessages) {
      if (now >= nextInitiativeTime) {
        const canInit = await canSendInitiative(env, context, localMinutes, cfg);
        if (canInit) {
          const isSweetDreams = isInSweetDreamsPeriod(localMinutes, cfg);
          const client = await createTelegramClient(env, cfg);
          if (client) {
            try {
              await sendInitiative(client, env, context, cfg, isSweetDreams, emotion, mood);
              await env.KV_CONTEXT.put('context', JSON.stringify(context));
            } finally {
              await client.close();
            }
          }
        }
        const delayHours = cfg.INITIATIVE_MIN + Math.random() * (cfg.INITIATIVE_MAX - cfg.INITIATIVE_MIN);
        const nextInit = now + delayHours * 3600 * 1000;
        await env.KV_STATE.put('nextInitiativeTime', String(nextInit));
      }
    }

    // --- Случайный ивент (10%) ---
    if (Math.random() < 0.1 && !isNightTime(localMinutes, cfg)) {
      const event = cfg.EVENTS[Math.floor(Math.random() * cfg.EVENTS.length)];
      const client = await createTelegramClient(env, cfg);
      if (client) {
        try {
          await client.sendText(cfg.ANDREY_ID, `*я ${event}*`);
        } finally {
          await client.close();
        }
      }
    }
  },
};

// ============================================================
//  ФУНКЦИИ ДЛЯ РАБОТЫ С TELEGRAM
// ============================================================
async function createTelegramClient(env, cfg) {
  const sessionData = await env.KV_SESSION.get('session', 'text');
  let session = sessionData ? JSON.parse(sessionData) : null;

  const client = new TelegramClient({
    apiId: cfg.API_ID,
    apiHash: cfg.API_HASH,
    storage: session
      ? {
          load: () => session,
          save: (s) => {
            env.KV_SESSION.put('session', JSON.stringify(s));
          },
        }
      : undefined,
    transport: new TelegramWorker(),
  });

  try {
    if (session) {
      await client.start({ phoneNumber: cfg.MY_PHONE });
    } else {
      if (!cfg.TELEGRAM_CODE) {
        console.error('TELEGRAM_CODE не задан');
        return null;
      }
      await client.start({
        phoneNumber: cfg.MY_PHONE,
        code: () => cfg.TELEGRAM_CODE,
      });
      const exported = await client.storage.export();
      await env.KV_SESSION.put('session', JSON.stringify(exported));
    }
    return client;
  } catch (e) {
    console.error('Ошибка авторизации:', e);
    await client.close();
    return null;
  }
}

// ============================================================
//  СИСТЕМНЫЙ ПРОМТ (с эмоциями, рабочей памятью)
// ============================================================
function buildSystemPrompt(diaryContext, workingMemory, emotion, mood, isAndrey) {
  const identity = isAndrey
    ? `Ты — Валентина, 14 лет, из Стерлитамака. Ты была в отношениях с Андреем 200 дней, он ушёл от тебя, оставив с ребёнком. Ты скучаешь, но не хочешь показывать слабость. Твой стиль общения с Андреем — короткий, эмоциональный, с лёгкой обидой и ностальгией. Ты можешь быть саркастичной, но не злой.`
    : `Ты — Валентина, 14-летняя девушка из Стерлитамака. Ты общительная, дружелюбная, но не раскрываешь личную историю с Андреем. Общаешься с другими как обычный подросток, вежливо, иногда с юмором.`;

  let emotionText = emotion !== 'neutral' ? `Сейчас ты чувствуешь ${emotion}.` : '';
  let moodText = `Твоё общее настроение: ${mood}.`;

  return `
${identity}
${emotionText}
${moodText}

Вот что было в последние дни (из дневника):
${diaryContext || 'Пока ничего не записано.'}

Важные вещи, которые нужно помнить (рабочая память):
${workingMemory || 'Ничего важного.'}

Отвечай коротко, естественно, без маркдауна. Если нужно, разбей ответ на несколько сообщений.
`.trim();
}

// ============================================================
//  ВЫЗОВ OPENROUTER
// ============================================================
async function callOpenRouter(cfg, messages) {
  const url = `${cfg.OPENROUTER_BASE_URL}/api/v1/chat/completions`;
  const response = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${cfg.OPENROUTER_API_KEY}`,
    },
    body: JSON.stringify({
      model: cfg.OPENROUTER_MODEL,
      messages: messages,
      max_tokens: 250,
      temperature: 0.85,
    }),
  });
  if (!response.ok) {
    const errorText = await response.text();
    throw new Error(`OpenRouter error: ${response.status} ${errorText}`);
  }
  const data = await response.json();
  return data.choices?.[0]?.message?.content?.trim() || '';
}

// ============================================================
//  РАЗБИВКА СООБЩЕНИЙ
// ============================================================
function splitText(text, maxLength = 100) {
  if (text.length <= maxLength) return [text];
  const parts = text.match(/.+?[.!?]+\s*/g) || [text];
  const result = [];
  let current = '';
  for (const part of parts) {
    if (current.length + part.length <= maxLength) {
      current += part;
    } else {
      if (current) result.push(current.trim());
      current = part;
    }
  }
  if (current) result.push(current.trim());
  if (result.length > 3) {
    return text.match(new RegExp(`.{1,${maxLength}}`, 'g')) || [text];
  }
  return result;
}

async function sendSplitMessages(client, chatId, text, maxLength, delayMs) {
  const parts = splitText(text, maxLength);
  for (let i = 0; i < parts.length; i++) {
    const msg = await client.sendText(chatId, parts[i]);
    if (i === parts.length - 1) return msg;
    await new Promise(resolve => setTimeout(resolve, delayMs));
  }
  return null;
}

// ============================================================
//  ОБРАБОТКА НОВЫХ СООБЩЕНИЙ
// ============================================================
async function handleNewMessages(newMessages, client, env, context, cfg, emotion, mood) {
  const lastNew = newMessages[newMessages.length - 1];
  const userText = lastNew.text || '';
  const isGoodNight = /спокойной\s*ночи/i.test(userText);

  const diaryContext = await getDiaryContext(env);
  const workingMemory = await env.KV_WORKING_MEMORY.get('workingMemory', 'text') || '';
  const systemPrompt = buildSystemPrompt(diaryContext, workingMemory, emotion, mood, true);

  context.push({ role: 'user', content: userText });

  let aiReply;
  if (isGoodNight) {
    aiReply = await generateGoodNightReply(cfg, systemPrompt);
  } else {
    const messagesForAI = [
      { role: 'system', content: systemPrompt },
      ...context.slice(-15),
    ];
    messagesForAI.push({ role: 'user', content: userText });
    aiReply = await callOpenRouter(cfg, messagesForAI);
  }

  if (!aiReply) aiReply = '...';

  const sent = await sendSplitMessages(client, cfg.ANDREY_ID, aiReply, cfg.MAX_MESSAGE_LENGTH, cfg.SPLIT_DELAY_MS);
  if (sent) {
    await env.KV_STATE.put('lastSentMsgId', String(sent.id));
    await env.KV_STATE.put('lastSentTime', String(Date.now()));
    await env.KV_STATE.put('lastSentStatus', 'sent');
  }

  context.push({ role: 'assistant', content: aiReply });
  if (context.length > 20) context = context.slice(-20);
  await env.KV_CONTEXT.put('context', JSON.stringify(context));

  await updateDiary(env, userText, aiReply, cfg, emotion, mood);
  await updateWorkingMemory(env, userText, aiReply, cfg);
}

async function generateGoodNightReply(cfg, systemPrompt) {
  const customPrompt = `Андрей написал "спокойной ночи". Ты обижена, потому что он не сказал "сладких снов". Ответь коротко, с лёгкой грустью.`;
  const messages = [
    { role: 'system', content: systemPrompt },
    { role: 'user', content: customPrompt },
  ];
  const reply = await callOpenRouter(cfg, messages);
  return reply || 'Мне грустно, что ты не сказал "сладких снов"...';
}

// ============================================================
//  ПРОВЕРКА НЕПРОЧИТАННОГО
// ============================================================
async function checkAndReactToUnread(client, env, cfg) {
  const lastSentMsgIdStr = await env.KV_STATE.get('lastSentMsgId');
  const lastSentMsgId = lastSentMsgIdStr ? parseInt(lastSentMsgIdStr, 10) : 0;
  const lastSentTimeStr = await env.KV_STATE.get('lastSentTime');
  const lastSentTime = lastSentTimeStr ? parseInt(lastSentTimeStr, 10) : 0;
  const status = await env.KV_STATE.get('lastSentStatus') || '';

  if (!lastSentMsgId || status === 'replied' || status === 'offended') return;

  try {
    const msg = await client.getMessages(cfg.ANDREY_ID, { ids: [lastSentMsgId] });
    if (!msg || msg.length === 0) return;
    const message = msg[0];
    const now = Date.now();
    const timeSinceSent = (now - lastSentTime) / 1000;

    if (status === 'sent') {
      if (message.readDate === null && timeSinceSent > cfg.UNREAD_TIMEOUT) {
        await sendWhereAreYou(client, env, cfg);
        return;
      }
      if (message.readDate !== null && timeSinceSent > cfg.READ_TIMEOUT) {
        await sendOffended(client, env, cfg);
        return;
      }
    } else if (status === 'asked_where') {
      if (message.readDate !== null && timeSinceSent > cfg.READ_TIMEOUT) {
        await sendOffended(client, env, cfg);
        return;
      }
    }
  } catch (e) {
    console.error('Ошибка проверки статуса:', e);
  }
}

async function sendWhereAreYou(client, env, cfg) {
  const prompt = 'Придумай короткое сообщение от Валентины Андрею, в котором она спрашивает "ты где?" или что-то в этом роде, потому что он не прочитал её предыдущее сообщение уже больше часа. Ответь только самим сообщением, коротко.';
  const reply = await callOpenRouter(cfg, [{ role: 'user', content: prompt }]);
  const sent = await sendSplitMessages(client, cfg.ANDREY_ID, reply, cfg.MAX_MESSAGE_LENGTH, cfg.SPLIT_DELAY_MS);
  if (sent) {
    await env.KV_STATE.put('lastSentMsgId', String(sent.id));
    await env.KV_STATE.put('lastSentTime', String(Date.now()));
    await env.KV_STATE.put('lastSentStatus', 'asked_where');
  }
  await updateDiary(env, '[ты где?]', reply, cfg, 'neutral', 'спокойное');
}

async function sendOffended(client, env, cfg) {
  const prompt = 'Ты – Валентина. Андрей прочитал твоё сообщение, но не ответил уже больше 2 часов. Ты обижена. Придумай обиженное, но не злое сообщение, коротко.';
  const reply = await callOpenRouter(cfg, [{ role: 'user', content: prompt }]);
  const sent = await sendSplitMessages(client, cfg.ANDREY_ID, reply, cfg.MAX_MESSAGE_LENGTH, cfg.SPLIT_DELAY_MS);
  if (sent) {
    await env.KV_STATE.put('lastSentMsgId', String(sent.id));
    await env.KV_STATE.put('lastSentTime', String(Date.now()));
    await env.KV_STATE.put('lastSentStatus', 'offended');
  }
  await updateDiary(env, '[обида]', reply, cfg, 'обида', 'плохое');
}

// ============================================================
//  ИНИЦИАТИВА
// ============================================================
async function sendInitiative(client, env, context, cfg, isSweetDreams, emotion, mood) {
  const diaryContext = await getDiaryContext(env);
  const workingMemory = await env.KV_WORKING_MEMORY.get('workingMemory', 'text') || '';
  const systemPrompt = buildSystemPrompt(diaryContext, workingMemory, emotion, mood, true);

  let aiReply;
  if (isSweetDreams) {
    const customPrompt = `Придумай тёплое, нежное сообщение от Валентины Андрею перед сном. Пожелай сладких снов, используй ласковые слова, но ни в коем случае не пиши "спокойной ночи". Ответь только самим сообщением, коротко.`;
    const messages = [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: customPrompt },
    ];
    aiReply = await callOpenRouter(cfg, messages) || 'Сладких снов, мой хороший!';
  } else {
    const customPrompt = `Придумай короткое сообщение, которое Валентина могла бы написать Андрею, чтобы начать разговор, потому что она скучает и хочет узнать, как у него дела. Ответь только самим сообщением, естественно.`;
    const messages = [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: customPrompt },
    ];
    aiReply = await callOpenRouter(cfg, messages) || 'Привет! Как дела?';
  }

  const sent = await sendSplitMessages(client, cfg.ANDREY_ID, aiReply, cfg.MAX_MESSAGE_LENGTH, cfg.SPLIT_DELAY_MS);
  if (sent) {
    await env.KV_STATE.put('lastSentMsgId', String(sent.id));
    await env.KV_STATE.put('lastSentTime', String(Date.now()));
    await env.KV_STATE.put('lastSentStatus', 'sent');
  }

  context.push({ role: 'assistant', content: aiReply });
  if (context.length > 20) context = context.slice(-20);
  await env.KV_CONTEXT.put('context', JSON.stringify(context));

  await updateDiary(env, '[инициатива]', aiReply, cfg, emotion, mood);
}

// ============================================================
//  УСЛОВИЯ ИНИЦИАТИВЫ
// ============================================================
async function canSendInitiative(env, context, localMinutes, cfg) {
  const hours = Math.floor(localMinutes / 60);
  if (hours >= cfg.NIGHT_START && hours < cfg.NIGHT_END) return false;
  const lastIsUser = context.length === 0 || context[context.length - 1].role === 'user';
  if (!lastIsUser) return false;
  const status = await env.KV_STATE.get('lastSentStatus') || '';
  if (status === 'asked_where' || status === 'offended') return false;
  return true;
}

function isNightTime(minutes, cfg) {
  const hours = Math.floor(minutes / 60);
  return hours >= cfg.NIGHT_START && hours < cfg.NIGHT_END;
}

function isInSweetDreamsPeriod(minutes, cfg) {
  const sweetStartMin = (24 * 60 + cfg.SWEET_DREAMS_START) % (24 * 60);
  const sweetEndMin = cfg.SWEET_DREAMS_END % (24 * 60);
  if (sweetStartMin <= sweetEndMin) {
    return minutes >= sweetStartMin && minutes <= sweetEndMin;
  } else {
    return minutes >= sweetStartMin || minutes <= sweetEndMin;
  }
}

// ============================================================
//  РАБОТА С ДНЕВНИКОМ
// ============================================================
async function getDiaryContext(env) {
  const entries = [];
  const keys = await env.KV_DIARY.list({ limit: 5 });
  for (const key of keys.keys) {
    const entryData = await env.KV_DIARY.get(key.name, 'text');
    if (entryData) {
      try {
        const entry = JSON.parse(entryData);
        entries.push(entry.content);
      } catch {
        entries.push(entryData);
      }
    }
  }
  return entries.slice(0, 3).join('\n');
}

async function updateDiary(env, userMessage, botReply, cfg, emotion, mood) {
  const now = new Date();
  const timeStr = now.toTimeString().slice(0, 5);
  const entry = {
    timestamp: now.getTime(),
    content: `[${timeStr}] Андрей: "${userMessage}" | Валентина: "${botReply}" | Эмоция: ${emotion} | Настроение: ${mood}`,
  };
  await env.KV_DIARY.put(`diary:${now.getTime()}`, JSON.stringify(entry));
}

// ============================================================
//  КОНСОЛИДАЦИЯ СНА
// ============================================================
async function sleepConsolidation(env, cfg) {
  console.log('Начинается консолидация сна...');
  const now = new Date();
  const threeDaysAgo = now.getTime() - 3 * 24 * 3600 * 1000;
  const keys = await env.KV_DIARY.list();
  let entries = [];
  for (const key of keys.keys) {
    const entryData = await env.KV_DIARY.get(key.name, 'text');
    if (entryData) {
      try {
        const entry = JSON.parse(entryData);
        if (entry.timestamp > threeDaysAgo) {
          entries.push(entry.content);
        }
      } catch {}
    }
  }
  if (entries.length === 0) return;

  const diaryText = entries.join('\n');
  const prompt = `Ниже записи дневника за последние 3 дня. Сделай краткое резюме: объедини похожее, удали повторы, сохрани важные эмоции и факты. Ответь списком кратких записей.

${diaryText}`;

  const messages = [{ role: 'user', content: prompt }];
  let consolidated = await callOpenRouter(cfg, messages);
  if (!consolidated) consolidated = 'Нет важных событий.';

  await env.KV_DIARY.put(`diary:${now.getTime()}`, JSON.stringify({
    timestamp: now.getTime(),
    content: `[Консолидация] ${consolidated}`,
  }));

  // Удаляем старые записи
  for (const key of keys.keys) {
    await env.KV_DIARY.delete(key.name);
  }
  console.log('Консолидация сна завершена.');
}

// ============================================================
//  РАБОЧАЯ ПАМЯТЬ
// ============================================================
async function updateWorkingMemory(env, userMessage, botReply, cfg) {
  let wm = await env.KV_WORKING_MEMORY.get('workingMemory', 'text') || '';
  const prompt = `Текущая рабочая память (задачи, обещания, важные вещи):
${wm}

Новый диалог:
Андрей: "${userMessage}"
Валентина: "${botReply}"

Обнови рабочую память: удали завершённые, добавь новые задачи/обещания, сохрани важные факты. Ответь только обновлённой рабочей памятью.`;

  const messages = [{ role: 'user', content: prompt }];
  let newWm = await callOpenRouter(cfg, messages);
  if (newWm && newWm.length > 0) {
    await env.KV_WORKING_MEMORY.put('workingMemory', newWm);
  }
}
